# Nginx 虚拟主机

## 编辑 `nginx.conf` 配置文件

```nginx
 sudo vim /usr/local/nginx/Nginx/conf/nginx.conf
```

##  添加虚拟域名配置文件

```nginx
include vhost/*.conf
```

## 配置虚拟主机

```nginx
server {
    listen 80;
    #listen [::]:80;
    server_name www.lushan.computer lushan.computer;
    return 301 https://lushan.computer$request_uri;
    index index.html index.htm index.php default.html default.htm default.php;
    root /home/wwwroot/wordpress;

    include rewrite/wordpress.conf;
    #error_page   404   /404.html;

    # Deny access to PHP files in specific directory
    #location ~ /(wp-content|uploads|wp-includes|images)/.*\.php$ { deny all; }

    include enable-php-pathinfo.conf;

    location ~ .*\.(gif|jpg|jpeg|png|bmp|swf)$ {
        expires 30d;
    }

    location ~ .*\.(js|css)?$ {
        expires 12h;
    }

    location ~ /.well-known {
        allow all;
    }

    location ~ /\. {
        deny all;
    }

    access_log /home/wwwlogs/lushan.computer.log;
}

server {
    listen 443 ssl http2;
    #listen [::]:443 ssl http2;
    server_name www.lushan.computer lushan.computer;
    index index.html index.htm index.php default.html default.htm default.php;
    root /home/wwwroot/wordpress;
    ssl on;
    ssl_certificate /usr/local/nginx/conf/ssl/lushan.computer/fullchain.cer;
    ssl_certificate_key /usr/local/nginx/conf/ssl/lushan.computer/lushan.computer.key;
    ssl_session_timeout 5m;
    ssl_protocols TLSv1 TLSv1.1 TLSv1.2;
    ssl_prefer_server_ciphers on;
    ssl_ciphers "EECDH+CHACHA20:EECDH+CHACHA20-draft:EECDH+AES128:RSA+AES128:EECDH+AES256:RSA+AES256:EECDH+3DES:RSA+3DES:!MD5";
    ssl_session_cache builtin:1000 shared:SSL:10m;
    # openssl dhparam -out /usr/local/nginx/conf/ssl/dhparam.pem 2048
    ssl_dhparam /usr/local/nginx/conf/ssl/dhparam.pem;

    include rewrite/wordpress.conf;
    #error_page   404   /404.html;

    # Deny access to PHP files in specific directory
    #location ~ /(wp-content|uploads|wp-includes|images)/.*\.php$ { deny all; }

    include enable-php-pathinfo.conf;

    location ~ .*\.(gif|jpg|jpeg|png|bmp|swf)$ {
        expires 30d;
    }

    location ~ .*\.(js|css)?$ {
        expires 12h;
    }

    location ~ /.well-known {
        allow all;
    }

    location ~ /\. {
        deny all;
    }

    access_log /home/wwwlogs/lushan.computer.log;
}
```
